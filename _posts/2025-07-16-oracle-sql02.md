### 집계함수로 데이터에서 값 불러오기

```sql
--select
select * from book;

--book 테이블에서 3500원인 가격의 도서명을 보시이오
select bookname
from book
where price = 35000;
--골프 바이블
-- 표에서 보면 실제 '골프바이블'책은 3500원 확인

--max: 최대값
--book: 테이블에서 가장 비싼 책을 출력하시오
select max(price) as 최대값
from book;
--가장비싼 책 : 골프바이블

--min: 최솟값
--book: 테이블에서 가장 저렴한 책을 출력하시오
select min(price)
from book;
--가장 저렴한 책 : 역도 단계별 기술

--count: 개수
--book 테이블에 저장된 책의 개수를 출력하시오
select count(*)as 개수
from book;
--책의 개수 : 10

--avg: 평균
select round(avg(price), 2) as 평균
from product;
--평균가격 : 23857.14 (소수점 2자리 까지 출력)

-- 책 가격 총합
SELECT SUM(price) as 상품가격총합
from product;
--총합 : 157500
